async function validateToken(token: string) {
    if (!token) return false;
    
    try {
        // Usamos fetch nativo para evitar conflitos de biblioteca no middleware
        const res = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/users/detail`, {
            headers: {
                Authorization: `Bearer ${token}`
            },
            cache: 'no-store' // Garante que ele sempre pergunte ao banco
        });

        if (!res.ok) return false;

        const data = await res.json();

        // VALIDAÇÃO SEGURA: 
        // Se data.isAdmin não existir ou for false, ele retorna false sem travar a tela
        return !!data?.isAdmin; 

    } catch (err) {
        console.error("Erro na validação do Middleware:", err);
        return false;
    }
}
